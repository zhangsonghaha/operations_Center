# 数据模型

### 数据库表结构

#### t_docker_container（Docker 容器表）

```sql
CREATE TABLE t_docker_container (
    container_id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '容器记录ID',
    docker_container_id VARCHAR(64) NOT NULL COMMENT 'Docker容器ID',
    container_name VARCHAR(100) NOT NULL COMMENT '容器名称',
    image_name VARCHAR(200) NOT NULL COMMENT '镜像名称',
    image_tag VARCHAR(50) DEFAULT 'latest' COMMENT '镜像标签',
    server_id BIGINT NOT NULL COMMENT '服务器ID',
    status VARCHAR(20) COMMENT '容器状态',
    port_mappings TEXT COMMENT '端口映射JSON',
    env_vars TEXT COMMENT '环境变量JSON',
    volumes TEXT COMMENT '卷挂载JSON',
    network_mode VARCHAR(50) COMMENT '网络模式',
    restart_policy VARCHAR(50) COMMENT '重启策略',
    cpu_limit VARCHAR(20) COMMENT 'CPU限制',
    memory_limit VARCHAR(20) COMMENT '内存限制',
    deploy_config TEXT COMMENT '完整部署配置JSON',
    template_id BIGINT COMMENT '使用的模板ID',
    deploy_log_id BIGINT COMMENT '部署日志ID',
    create_by VARCHAR(64) COMMENT '创建者',
    create_time DATETIME COMMENT '创建时间',
    update_by VARCHAR(64) COMMENT '更新者',
    update_time DATETIME COMMENT '更新时间',
    remark VARCHAR(500) COMMENT '备注',
    INDEX idx_server_id (server_id),
    INDEX idx_docker_container_id (docker_container_id),
    INDEX idx_status (status),
    FOREIGN KEY (server_id) REFERENCES t_ops_server(server_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Docker容器管理表';
```

#### t_docker_template（Docker 模板表）

```sql
CREATE TABLE t_docker_template (
    template_id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '模板ID',
    template_name VARCHAR(100) NOT NULL COMMENT '模板名称',
    template_type VARCHAR(50) NOT NULL COMMENT '模板类型',
    description VARCHAR(500) COMMENT '模板描述',
    image_name VARCHAR(200) NOT NULL COMMENT '镜像名称',
    default_tag VARCHAR(50) DEFAULT 'latest' COMMENT '默认标签',
    config_template TEXT NOT NULL COMMENT '配置模板JSON',
    is_predefined CHAR(1) DEFAULT '0' COMMENT '是否预定义(0否 1是)',
    icon VARCHAR(200) COMMENT '图标URL',
    sort_order INT DEFAULT 0 COMMENT '排序',
    status CHAR(1) DEFAULT '0' COMMENT '状态(0正常 1停用)',
    create_by VARCHAR(64) COMMENT '创建者',
    create_time DATETIME COMMENT '创建时间',
    update_by VARCHAR(64) COMMENT '更新者',
    update_time DATETIME COMMENT '更新时间',
    INDEX idx_template_type (template_type),
    INDEX idx_is_predefined (is_predefined)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Docker部署模板表';
```

#### t_docker_image（Docker 镜像缓存表）

```sql
CREATE TABLE t_docker_image (
    image_id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '镜像记录ID',
    server_id BIGINT NOT NULL COMMENT '服务器ID',
    image_name VARCHAR(200) NOT NULL COMMENT '镜像名称',
    image_tag VARCHAR(50) NOT NULL COMMENT '镜像标签',
    image_digest VARCHAR(100) COMMENT '镜像摘要',
    size_mb DECIMAL(10,2) COMMENT '镜像大小(MB)',
    created_at DATETIME COMMENT '镜像创建时间',
    pulled_at DATETIME COMMENT '拉取时间',
    last_used_at DATETIME COMMENT '最后使用时间',
    INDEX idx_server_id (server_id),
    INDEX idx_image_name_tag (image_name, image_tag),
    FOREIGN KEY (server_id) REFERENCES t_ops_server(server_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='Docker镜像缓存表';
```

### 领域模型

#### DockerDeployConfig

```java
public class DockerDeployConfig {
    private Long serverId;              // 服务器ID
    private String containerName;       // 容器名称
    private String imageName;           // 镜像名称
    private String imageTag;            // 镜像标签
    private List<PortMapping> ports;    // 端口映射
    private Map<String, String> envVars; // 环境变量
    private List<VolumeMount> volumes;  // 卷挂载
    private String networkMode;         // 网络模式
    private String restartPolicy;       // 重启策略
    private ResourceLimit resources;    // 资源限制
    private Long templateId;            // 模板ID
    private String remark;              // 备注
}
```

#### PortMapping

```java
public class PortMapping {
    private Integer hostPort;           // 主机端口
    private Integer containerPort;      // 容器端口
    private String protocol;            // 协议(tcp/udp)
}
```

#### VolumeMount

```java
public class VolumeMount {
```
