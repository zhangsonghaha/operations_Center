# 关键算法与实现

### 算法 1: Docker 环境检测

**目的**: 检测远程服务器的 Docker 环境是否可用

```pascal
ALGORITHM checkDockerEnvironment(serverId)
INPUT: serverId - 服务器ID
OUTPUT: DockerEnvInfo - Docker环境信息

PRECONDITIONS:
  - serverId 存在且有效
  - 服务器SSH连接可用

POSTCONDITIONS:
  - 返回完整的Docker环境信息
  - 如果Docker未安装，dockerInstalled = false
  - 如果检测失败，抛出异常

BEGIN
  // 步骤1: 获取SSH会话
  session ← getSSHSession(serverId)
  envInfo ← new DockerEnvInfo()
  
  // 步骤2: 检查Docker是否安装
  TRY
    versionOutput ← executeCommand(session, "docker --version")
    envInfo.dockerInstalled ← true
    envInfo.dockerVersion ← parseVersion(versionOutput)
  CATCH Exception
    envInfo.dockerInstalled ← false
    RETURN envInfo
  END TRY
  
  // 步骤3: 检查Docker是否运行
  TRY
    executeCommand(session, "docker info")
    envInfo.dockerRunning ← true
  CATCH Exception
    envInfo.dockerRunning ← false
    RETURN envInfo
  END TRY
  
  // 步骤4: 获取Docker详细信息
  infoOutput ← executeCommand(session, "docker info --format '{{json .}}'")
  infoJson ← parseJSON(infoOutput)
  
  envInfo.dockerRootDir ← infoJson.get("DockerRootDir")
  envInfo.storageDriver ← infoJson.get("Driver")
  
  // 步骤5: 获取磁盘空间信息
  dfOutput ← executeCommand(session, "df -B1 " + envInfo.dockerRootDir)
  spaceInfo ← parseDiskSpace(dfOutput)
  envInfo.totalSpace ← spaceInfo.total
  envInfo.usedSpace ← spaceInfo.used
  
  RETURN envInfo
END

LOOP INVARIANTS:
  - session 保持有效连接
  - envInfo 对象状态一致
```

### 算法 2: 容器一键部署

**目的**: 执行完整的容器部署流程

```pascal
ALGORITHM deployContainer(config)
INPUT: config - Docker部署配置
OUTPUT: DockerDeployResult - 部署结果

PRECONDITIONS:
  - config 非空且已验证
  - config.serverId 有效
  - config.imageName 非空
  - config.containerName 非空且唯一

POSTCONDITIONS:
  - 容器成功创建并运行，返回容器ID
  - 部署日志已记录
  - 如果失败，回滚已创建的资源

BEGIN
  // 步骤1: 初始化
  session ← getSSHSession(config.serverId)
  logId ← createDeployLog(config)
  result ← new DockerDeployResult()
  
  TRY
    // 步骤2: 环境检测
    pushLog(logId, "开始检测Docker环境...")
    envInfo ← checkDockerEnvironment(config.serverId)
    
    IF NOT envInfo.dockerInstalled OR NOT envInfo.dockerRunning THEN
      THROW Exception("Docker环境不可用")
    END IF
    pushLog(logId, "Docker环境检测通过")
    
    // 步骤3: 检查容器名冲突
    pushLog(logId, "检查容器名称冲突...")
    existingContainers ← executeCommand(session, 
      "docker ps -a --filter name=" + config.containerName + " --format '{{.Names}}'")
    
    IF existingContainers CONTAINS config.containerName THEN
      THROW Exception("容器名称已存在: " + config.containerName)
    END IF
    
    // 步骤4: 检查端口冲突
    pushLog(logId, "检查端口冲突...")
    FOR EACH portMapping IN config.ports DO
      isPortUsed ← checkPortInUse(session, portMapping.hostPort)
      IF isPortUsed THEN
        THROW Exception("端口已被占用: " + portMapping.hostPort)
      END IF
    END FOR
    
    // 步骤5: 拉取镜像
    imageFullName ← config.imageName + ":" + config.imageTag
    pushLog(logId, "开始拉取镜像: " + imageFullName)
    
    pullSuccess ← pullImageWithProgress(session, config.imageName, 
                                        config.imageTag, logId)
    IF NOT pullSuccess THEN
      THROW Exception("镜像拉取失败")
    END IF
    pushLog(logId, "镜像拉取完成")
    
    // 步骤6: 构建docker run命令
    pushLog(logId, "构建容器启动命令...")
    dockerCommand ← buildDockerRunCommand(config)
    pushLog(logId, "执行命令: " + dockerCommand)
    
    // 步骤7: 创建并启动容器
    pushLog(logId, "创建并启动容器...")
    containerIdOutput ← executeCommand(session, dockerCommand)
    containerId ← trim(containerIdOutput)
    
    IF isEmpty(containerId) OR length(containerId) != 64 THEN
      THROW Exception("容器创建失败，未返回有效容器ID")
    END IF
    
    result.containerId ← containerId
    pushLog(logId, "容器创建成功: " + substring(containerId, 0, 12))
    
    // 步骤8: 健康检查
    pushLog(logId, "执行健康检查...")
    healthCheckPassed ← performHealthCheck(session, containerId, 30)
    
    IF NOT healthCheckPassed THEN
      pushLog(logId, "警告: 健康检查未通过，但容器已启动")
    ELSE
      pushLog(logId, "健康检查通过")
    END IF
    
    // 步骤9: 保存容器记录
    containerRecord ← createContainerRecord(config, containerId, logId)
    saveContainerRecord(containerRecord)
    
    // 步骤10: 更新镜像缓存
    updateImageCache(config.serverId, config.imageName, config.imageTag)
    
    result.success ← true
    result.message ← "容器部署成功"
    pushLog(logId, "部署完成")
    updateDeployLogStatus(logId, "SUCCESS")
    
  CATCH Exception e
    // 回滚处理
    pushLog(logId, "部署失败: " + e.message)
    
    IF result.containerId IS NOT NULL THEN
      pushLog(logId, "尝试清理已创建的容器...")
      TRY
        executeCommand(session, "docker rm -f " + result.containerId)
        pushLog(logId, "容器已清理")
      CATCH cleanupException
        pushLog(logId, "容器清理失败: " + cleanupException.message)
      END TRY
    END IF
    
    result.success ← false
    result.message ← e.message
    updateDeployLogStatus(logId, "FAILED")
    
  FINALLY
    closeSSHSession(session)
  END TRY
  
  RETURN result
END

LOOP INVARIANTS:
  - 在FOR循环中，所有已检查的端口未被占用
  - session 保持有效连接直到FINALLY块
  - logId 始终有效且可用于日志推送
```

### 算法 3: 构建 Docker Run 命令

**目的**: 根据配置生成完整的 docker run 命令

```pascal
ALGORITHM buildDockerRunCommand(config)
INPUT: config - Docker部署配置
OUTPUT: String - 完整的docker run命令

PRECONDITIONS:
  - config 非空
  - config.containerName 非空
  - config.imageName 非空

POSTCONDITIONS:
  - 返回有效的docker run命令字符串
  - 命令包含所有必要的参数

BEGIN
  command ← "docker run -d"
  
  // 添加容器名称
  command ← command + " --name " + escapeShellArg(config.containerName)
  
  // 添加端口映射
  FOR EACH portMapping IN config.ports DO
    ASSERT portMapping.hostPort > 0 AND portMapping.containerPort > 0
    
    portStr ← portMapping.hostPort + ":" + portMapping.containerPort
    IF portMapping.protocol IS NOT NULL THEN
      portStr ← portStr + "/" + portMapping.protocol
    END IF
    command ← command + " -p " + portStr
  END FOR
  
  // 添加环境变量
  FOR EACH (key, value) IN config.envVars DO
    ASSERT key IS NOT NULL AND NOT isEmpty(key)
    
    escapedValue ← escapeShellArg(value)
    command ← command + " -e " + key + "=" + escapedValue
  END FOR
  
  // 添加卷挂载
  FOR EACH volumeMount IN config.volumes DO
    ASSERT volumeMount.hostPath IS NOT NULL
    ASSERT volumeMount.containerPath IS NOT NULL
    
    volumeStr ← escapeShellArg(volumeMount.hostPath) + ":" + 
                volumeMount.containerPath
    
    IF volumeMount.mode IS NOT NULL THEN
      volumeStr ← volumeStr + ":" + volumeMount.mode
    END IF
    
    command ← command + " -v " + volumeStr
  END FOR
  
  // 添加网络模式
  IF config.networkMode IS NOT NULL THEN
    command ← command + " --network " + config.networkMode
  END IF
  
  // 添加重启策略
  IF config.restartPolicy IS NOT NULL THEN
    command ← command + " --restart " + config.restartPolicy
  ELSE
    command ← command + " --restart unless-stopped"
  END IF
  
  // 添加资源限制
  IF config.resources IS NOT NULL THEN
    IF config.resources.cpuLimit IS NOT NULL THEN
      command ← command + " --cpus " + config.resources.cpuLimit
    END IF
    
    IF config.resources.memoryLimit IS NOT NULL THEN
      command ← command + " --memory " + config.resources.memoryLimit
    END IF
  END IF
  
  // 添加镜像名称和标签
  imageFullName ← config.imageName + ":" + config.imageTag
  command ← command + " " + imageFullName
  
  RETURN command
END

LOOP INVARIANTS:
  - command 字符串始终以 "docker run -d" 开头
  - 所有添加的参数格式正确
  - 在端口映射循环中，所有端口号为正整数
  - 在环境变量循环中，所有key非空
  - 在卷挂载循环中，所有路径非空
```


### 算法 4: 镜像拉取与进度监控

**目的**: 拉取 Docker 镜像并实时推送进度

```pascal
ALGORITHM pullImageWithProgress(session, imageName, imageTag, logId)
INPUT: session - SSH会话, imageName - 镜像名, imageTag - 标签, logId - 日志ID
OUTPUT: boolean - 是否成功

PRECONDITIONS:
  - session 有效且已连接
  - imageName 非空
  - imageTag 非空
  - logId 有效

POSTCONDITIONS:
  - 镜像拉取完成或失败
  - 进度日志已推送
  - 返回true表示成功，false表示失败

BEGIN
  imageFullName ← imageName + ":" + imageTag
  
  // 检查镜像是否已存在
  checkCommand ← "docker images -q " + imageFullName
  existingImage ← executeCommand(session, checkCommand)
  
  IF NOT isEmpty(existingImage) THEN
    pushLog(logId, "镜像已存在，跳过拉取")
    RETURN true
  END IF
  
  // 执行拉取命令
  pullCommand ← "docker pull " + imageFullName
  channel ← session.openChannel("exec")
  channel.setCommand(pullCommand)
  
  inputStream ← channel.getInputStream()
  channel.connect()
  
  buffer ← new byte[1024]
  lastProgress ← ""
  
  // 实时读取输出
  WHILE channel.isConnected() OR inputStream.available() > 0 DO
    ASSERT channel IS NOT NULL
    
    IF inputStream.available() > 0 THEN
      bytesRead ← inputStream.read(buffer)
      IF bytesRead > 0 THEN
        output ← new String(buffer, 0, bytesRead)
        
        // 解析进度信息
        lines ← split(output, "\n")
        FOR EACH line IN lines DO
          IF contains(line, "Pulling") OR contains(line, "Downloading") OR
             contains(line, "Extracting") OR contains(line, "Pull complete") THEN
            
            IF line != lastProgress THEN
              pushLog(logId, line)
              lastProgress ← line
            END IF
          END IF
        END FOR
      END IF
    ELSE
      sleep(100)
    END IF
  END WHILE
  
  exitStatus ← channel.getExitStatus()
  channel.disconnect()
  
  IF exitStatus = 0 THEN
    pushLog(logId, "镜像拉取成功: " + imageFullName)
    RETURN true
  ELSE
    pushLog(logId, "镜像拉取失败，退出码: " + exitStatus)
    RETURN false
  END IF
END

LOOP INVARIANTS:
  - channel 保持连接状态直到数据读取完成
  - inputStream 可用且未关闭
  - 所有推送的日志行不重复（通过lastProgress控制）
```

### 算法 5: 容器健康检查

**目的**: 检查容器是否成功启动并运行正常

```pascal
ALGORITHM performHealthCheck(session, containerId, timeoutSeconds)
INPUT: session - SSH会话, containerId - 容器ID, timeoutSeconds - 超时时间
OUTPUT: boolean - 健康检查是否通过

PRECONDITIONS:
  - session 有效且已连接
  - containerId 非空且有效
  - timeoutSeconds > 0

POSTCONDITIONS:
  - 返回true表示容器运行正常
  - 返回false表示容器未正常运行或超时
  - 不抛出异常

BEGIN
  startTime ← currentTimeMillis()
  checkInterval ← 2000  // 2秒检查一次
  
  WHILE (currentTimeMillis() - startTime) < (timeoutSeconds * 1000) DO
    ASSERT session.isConnected()
    
    TRY
      // 检查容器状态
      statusCommand ← "docker inspect --format='{{.State.Status}}' " + containerId
      status ← trim(executeCommand(session, statusCommand))
      
      IF status = "running" THEN
        // 容器正在运行，进一步检查健康状态
        healthCommand ← "docker inspect --format='{{.State.Health.Status}}' " + 
                       containerId
        healthStatus ← trim(executeCommand(session, healthCommand))
        
        // 如果没有健康检查配置，返回true
        IF healthStatus = "<no value>" OR isEmpty(healthStatus) THEN
          RETURN true
        END IF
        
        // 如果有健康检查，等待健康状态为healthy
        IF healthStatus = "healthy" THEN
          RETURN true
        ELSE IF healthStatus = "unhealthy" THEN
          RETURN false
        END IF
        // 如果是starting状态，继续等待
        
      ELSE IF status = "exited" OR status = "dead" THEN
        // 容器已退出或死亡
        RETURN false
      END IF
      
    CATCH Exception e
      // 忽略异常，继续重试
    END TRY
    
    sleep(checkInterval)
  END WHILE
  
  // 超时
  RETURN false
END

LOOP INVARIANTS:
  - 已等待时间 ≤ timeoutSeconds
  - session 保持连接状态
  - 每次循环间隔 checkInterval 毫秒
```

### 算法 6: 端口冲突检查

**目的**: 检查主机端口是否已被占用

```pascal
ALGORITHM checkPortInUse(session, port)
INPUT: session - SSH会话, port - 端口号
OUTPUT: boolean - 端口是否被占用

PRECONDITIONS:
  - session 有效且已连接
  - port > 0 AND port < 65536

POSTCONDITIONS:
  - 返回true表示端口已被占用
  - 返回false表示端口可用

BEGIN
  ASSERT port > 0 AND port < 65536
  
  // 使用netstat或ss命令检查端口
  // 优先使用ss（更快）
  TRY
    command ← "ss -tuln | grep ':" + port + " '"
    output ← executeCommand(session, command)
    
    IF NOT isEmpty(output) THEN
      RETURN true
    END IF
    
  CATCH Exception e
    // ss命令不可用，尝试netstat
    TRY
      command ← "netstat -tuln | grep ':" + port + " '"
      output ← executeCommand(session, command)
      
      IF NOT isEmpty(output) THEN
        RETURN true
      END IF
      
    CATCH Exception e2
      // 两个命令都不可用，尝试lsof
      TRY
        command ← "lsof -i :" + port
        output ← executeCommand(session, command)
        
        IF NOT isEmpty(output) THEN
          RETURN true
        END IF
        
      CATCH Exception e3
        // 无法检查，假设端口可用
        RETURN false
      END TRY
    END TRY
  END TRY
  
  RETURN false
END
```
